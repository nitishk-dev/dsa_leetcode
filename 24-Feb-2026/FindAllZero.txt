public class AllZeroFromStart {
  public static int FindAllZero(int[] arr){
    for(int i = 0; i < arr.length - 1; i++){
      if(arr[i] == 0) continue;

      if(arr[i+1] != arr[i] && arr[i + 1] != arr[i] + 1){
        return arr[i] + 1;
      }
    }
    return -1;
  }

  public static void main(String[] args){
    int[] arr = {5, 8, 4, 0, 66, 0, 55, 0, 44, 0, 41};;
    int allZero = FindAllZero(arr);
    System.out.println(allZero);
  }
}
