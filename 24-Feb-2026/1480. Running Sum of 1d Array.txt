/*
    ====Pseudocode====
    1) First we create new arr called *result*
    2) We take a variable *sum* to store the running total
    3) A loop runs from the beginning of the array to the end
    4) At each step:
       i) We add the current number to *sum*
       ii) We store the updated *sum* in the *result* array.
    5) At the end, we return the result array

    ===Dry Run===
    index 0 -> 1;
    index 1 -> 1 + 2 = 3
    index 2 -> 1 + 2 + 3 = 6;
    index 3 -> 1 + 2 + 3 + 4 = 10
*/



class Solution {
    public int[] runningSum(int[] nums) {
        int[] result = new int[nums.length];
        int sum = 0;

        for(int i = 0; i < nums.length; i++){
            sum += nums[i];
            result[i] = sum;
        }
        return result;
    }
}
