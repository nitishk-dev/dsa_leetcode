/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

 
/*
    // Pseudocode
    * Start from rool
    * While current node it not null;
      * if current node value equals, return current node
      * if target is less than current node value, move to left child
      * Else move to right child
    * If loop ends, return null
*/


class Solution {
    public TreeNode searchBST(TreeNode root, int val) {
        while (root != null){
            if(root.val == val){
                return root;
            }else if(val < root.val){
                root = root.left;
            }else{
                root = root.right;
            }
        }
        return null;
    }
}
